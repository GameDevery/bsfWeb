83  class BS_CORE_EXPORT PrefabDiffRTTI : public RTTIType < PrefabDiff, IReflectable, PrefabDiffRTTI >
84  {
86  struct SerializedHandle
87  {
88  SPtr<SerializedObject> object;
89  SPtr<GameObjectHandleBase> handle;
90  };
91 
92  private:
93  BS_BEGIN_RTTI_MEMBERS
94  BS_RTTI_MEMBER_REFLPTR(mRoot, 0)
95  BS_END_RTTI_MEMBERS
96  public:
97  void onDeserializationStarted(IReflectable* obj, const UnorderedMap<String, UINT64>& params) override
98  {
99  PrefabDiff* prefabDiff = static_cast<PrefabDiff*>(obj);
100 
101  if (GameObjectManager::instance().isGameObjectDeserializationActive())
102  GameObjectManager::instance().registerOnDeserializationEndCallback(std::bind(&PrefabDiffRTTI::delayedOnDeserializationEnded, prefabDiff));
103  }
104 
105  void onDeserializationEnded(IReflectable* obj, const UnorderedMap<String, UINT64>& params) override
106  {
107  assert(GameObjectManager::instance().isGameObjectDeserializationActive());
108 
109  // Make sure to deserialize all game object handles since their IDs need to be updated. Normally they are
110  // updated automatically upon deserialization but since we store them in intermediate form we need to manually
111  // deserialize and reserialize them in order to update their IDs.
112  PrefabDiff* prefabDiff = static_cast<PrefabDiff*>(obj);
113 
114  Stack<SPtr<PrefabObjectDiff>> todo;
115 
116  if (prefabDiff->mRoot != nullptr)
117  todo.push(prefabDiff->mRoot);
118 
119  UnorderedSet<SPtr<SerializedObject>> handleObjects;
120 
121  while (!todo.empty())
122  {
123  SPtr<PrefabObjectDiff> current = todo.top();
124  todo.pop();
125 
126  for (auto& component : current->addedComponents)
127  findGameObjectHandles(component, handleObjects);
128 
129  for (auto& child : current->addedChildren)
130  findGameObjectHandles(child, handleObjects);
131 
132  for (auto& component : current->componentDiffs)
133  findGameObjectHandles(component->data, handleObjects);
134 
135  for (auto& child : current->childDiffs)
136  todo.push(child);
137  }
138 
139  Vector<SerializedHandle> handleData(handleObjects.size());
140 
141  UINT32 idx = 0;
142  BinarySerializer bs;
143  for (auto& handleObject : handleObjects)
144  {
145  SerializedHandle& handle = handleData[idx];
146 
147  handle.object = handleObject;
148  handle.handle = std::static_pointer_cast<GameObjectHandleBase>(bs._decodeFromIntermediate(handleObject));
149 
150  idx++;
151  }
152 
153  prefabDiff->mRTTIData = handleData;
154  }
155 
161  static void delayedOnDeserializationEnded(PrefabDiff* prefabDiff)
162  {
163  Vector<SerializedHandle>& handleData = any_cast_ref<Vector<SerializedHandle>>(prefabDiff->mRTTIData);
164 
165  BinarySerializer bs;
166  for (auto& serializedHandle : handleData)
167  {
168  if (serializedHandle.handle != nullptr)
169  *serializedHandle.object = *bs._encodeToIntermediate(serializedHandle.handle.get());
170  }
171 
172  prefabDiff->mRTTIData = nullptr;
173  }
174 
176  static void findGameObjectHandles(const SPtr<SerializedObject>& serializedObject, UnorderedSet<SPtr<SerializedObject>>& handleObjects)
177  {
178  for (auto& subObject : serializedObject->subObjects)
179  {
180  RTTITypeBase* rtti = IReflectable::_getRTTIfromTypeId(subObject.typeId);
181  if (rtti == nullptr)
182  continue;
183 
184  if (rtti->getRTTIId() == TID_GameObjectHandleBase)
185  {
186  handleObjects.insert(serializedObject);
187  return;
188  }
189 
190  for (auto& child : subObject.entries)
191  {
192  RTTIField* curGenericField = rtti->findField(child.second.fieldId);
193  if (curGenericField == nullptr)
194  continue;
195 
196  SPtr<SerializedInstance> entryData = child.second.serialized;
197  if (entryData == nullptr)
198  continue;
199 
200  if (rtti_is_of_type<SerializedArray>(entryData))
201  {
202  SPtr<SerializedArray> arrayData = std::static_pointer_cast<SerializedArray>(entryData);
203 
204  for (auto& arrayElem : arrayData->entries)
205  {
206  if (arrayElem.second.serialized != nullptr && rtti_is_of_type<SerializedObject>(arrayElem.second.serialized))
207  {
208  SPtr<SerializedObject> arrayElemData = std::static_pointer_cast<SerializedObject>(arrayElem.second.serialized);
209  findGameObjectHandles(arrayElemData, handleObjects);
210  }
211  }
212  }
213  else if(rtti_is_of_type<SerializedObject>(entryData))
214  {
215  SPtr<SerializedObject> fieldObjectData = std::static_pointer_cast<SerializedObject>(entryData);
216  findGameObjectHandles(fieldObjectData, handleObjects);
217  }
218  }
219  }
220  }
