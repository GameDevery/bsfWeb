283  template<class T>
284  void parseWindows(const T* pathStr, UINT32 numChars)
285  {
286  clear();
287 
288  UINT32 idx = 0;
289  BasicStringStream<T> tempStream;
290 
291  if (idx < numChars)
292  {
293  if (pathStr[idx] == '\\' || pathStr[idx] == '/')
294  {
295  mIsAbsolute = true;
296  idx++;
297  }
298  }
299 
300  if (idx < numChars)
301  {
302  // Path starts with a node, a drive letter or is relative
303  if (mIsAbsolute && (pathStr[idx] == '\\' || pathStr[idx] == '/')) // Node
304  {
305  idx++;
306 
307  tempStream.str(BasicString<T>());
308  tempStream.clear();
309  while (idx < numChars && pathStr[idx] != '\\' && pathStr[idx] != '/')
310  tempStream << pathStr[idx++];
311 
312  setNode(tempStream.str());
313 
314  if (idx < numChars)
315  idx++;
316  }
317  else // A drive letter or not absolute
318  {
319  T drive = pathStr[idx];
320  idx++;
321 
322  if (idx < numChars && pathStr[idx] == ':')
323  {
324  if (mIsAbsolute || !((drive >= 'a' && drive <= 'z') || (drive >= 'A' && drive <= 'Z')))
325  throwInvalidPathException(BasicString<T>(pathStr, numChars));
326 
327  mIsAbsolute = true;
328  setDevice(String(1, drive));
329 
330  idx++;
331 
332  if (idx >= numChars || (pathStr[idx] != '\\' && pathStr[idx] != '/'))
333  throwInvalidPathException(BasicString<T>(pathStr, numChars));
334 
335  idx++;
336  }
337  else
338  idx--;
339  }
340 
341  while (idx < numChars)
342  {
343  tempStream.str(BasicString<T>());
344  tempStream.clear();
345  while (idx < numChars && pathStr[idx] != '\\' && pathStr[idx] != '/')
346  {
347  tempStream << pathStr[idx];
348  idx++;
349  }
350 
351  if (idx < numChars)
352  pushDirectory(tempStream.str());
353  else
354  setFilename(tempStream.str());
355 
356  idx++;
357  }
358  }
359  }

362  template<class T>
363  void parseUnix(const T* pathStr, UINT32 numChars)
364  {
365  clear();
366 
367  UINT32 idx = 0;
368  BasicStringStream<T> tempStream;
369 
370  if (idx < numChars)
371  {
372  if (pathStr[idx] == '/')
373  {
374  mIsAbsolute = true;
375  idx++;
376  }
377  else if (pathStr[idx] == '~')
378  {
379  idx++;
380  if (idx >= numChars || pathStr[idx] == '/')
381  {
382  pushDirectory(String("~"));
383  mIsAbsolute = true;
384  }
385  else
386  idx--;
387  }
388 
389  while (idx < numChars)
390  {
391  tempStream.str(BasicString<T>());
392  tempStream.clear();
393  while (idx < numChars && pathStr[idx] != '/')
394  {
395  tempStream << pathStr[idx];
396  idx++;
397  }
398 
399  if (idx < numChars)
400  {
401  if (mDirectories.empty())
402  {
403  BasicString<T> deviceStr = tempStream.str();
404  if (!deviceStr.empty() && *(deviceStr.rbegin()) == ':')
405  {
406  setDevice(deviceStr.substr(0, deviceStr.length() - 1));
407  mIsAbsolute = true;
408  }
409  else
410  {
411  pushDirectory(deviceStr);
412  }
413  }
414  else
415  {
416  pushDirectory(tempStream.str());
417  }
418  }
419  else
420  {
421  setFilename(tempStream.str());
422  }
423 
424  idx++;
425  }
426  }
427  }

479  {
480  UINT32 size;
481  memcpy(&size, memory, sizeof(UINT32));
482  memory += sizeof(UINT32);
483 
484  memory = rttiReadElem(data.mDevice, memory);
485  memory = rttiReadElem(data.mNode, memory);
486  memory = rttiReadElem(data.mFilename, memory);
487  memory = rttiReadElem(data.mIsAbsolute, memory);
488  rttiReadElem(data.mDirectories, memory);

493  static UINT32 getDynamicSize(const Path& data)
494  {
495  UINT64 dataSize = rttiGetElemSize(data.mDevice) + rttiGetElemSize(data.mNode) + rttiGetElemSize(data.mFilename) +
496  rttiGetElemSize(data.mIsAbsolute) + rttiGetElemSize(data.mDirectories) + sizeof(UINT32);
497 
498 #if BS_DEBUG_MODE

501  __string_throwDataOverflowException();

503 #endif
504 
505  return (UINT32)dataSize;
