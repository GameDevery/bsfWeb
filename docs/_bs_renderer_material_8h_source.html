176  void bind() const
177  {
178  RenderAPI& rapi = RenderAPI::instance();
179 
180  if(mGfxPipeline)
181  {
182  rapi.setGraphicsPipeline(mGfxPipeline);
183  rapi.setStencilRef(mStencilRef);
184  }
185  else
186  rapi.setComputePipeline(mComputePipeline);
187 
188  rapi.setGpuParams(mParams);
189  }
190 
191  protected:
192  RendererMaterial()
193  {
194  mInitOnStart.instantiate();
195 
196  if(mMetaData.overrideShader)
197  mShader = mMetaData.overrideShader;
198  else
199  mShader = mMetaData.shader;
200 
201  mVariation = mMetaData.variations.get(_varIdx);
202 
203  const Vector<SPtr<Technique>>& techniques = mShader->getTechniques();
204  for(auto& entry : techniques)
205  {
206  if(!entry->isSupported())
207  continue;
208 
209  if(entry->getVariation() == mVariation)
210  {
211  SPtr<Pass> pass = entry->getPass(0);
212  pass->compile();
213 
214  mGfxPipeline = pass->getGraphicsPipelineState();
215  if (mGfxPipeline != nullptr)
216  mParams = GpuParams::create(mGfxPipeline);
217  else
218  {
219  mComputePipeline = pass->getComputePipelineState();
220  mParams = GpuParams::create(mComputePipeline);
221  }
222 
223  // Assign default values from the shader
224  const auto& textureParams = mShader->getTextureParams();
225  for(auto& param : textureParams)
226  {
227  UINT32 defaultValueIdx = param.second.defaultValueIdx;
228  if(defaultValueIdx == (UINT32)-1)
229  continue;
230 
231  for (UINT32 i = 0; i < 6; i++)
232  {
233  GpuProgramType progType = (GpuProgramType)i;
234 
235  for(auto& varName : param.second.gpuVariableNames)
236  {
237  if(mParams->hasTexture(progType, varName))
238  {
239  SPtr<Texture> texture = mShader->getDefaultTexture(defaultValueIdx);
240  mParams->setTexture(progType, varName, texture);
241  }
242  }
243  }
244  }
245 
246  const auto& samplerParams = mShader->getSamplerParams();
247  for(auto& param : samplerParams)
248  {
249  UINT32 defaultValueIdx = param.second.defaultValueIdx;
250  if(defaultValueIdx == (UINT32)-1)
251  continue;
252 
253  for (UINT32 i = 0; i < 6; i++)
254  {
255  GpuProgramType progType = (GpuProgramType)i;
256 
257  for(auto& varName : param.second.gpuVariableNames)
258  {
259  if(mParams->hasSamplerState(progType, varName))
260  {
261  SPtr<SamplerState> samplerState = mShader->getDefaultSampler(defaultValueIdx);
262  mParams->setSamplerState(progType, varName, samplerState);
263  }
264  }
265  }
266  }
267 
268  mStencilRef = pass->getStencilRefValue();
269  }
270  }
271  }
