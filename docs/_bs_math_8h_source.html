513  template <typename T>
514  static UINT32 solveQuadratic(T A, T B, T C, T* roots)
515  {
516  if (!approxEquals(A, (T)0))
517  {
518  T p = B / (2 * A);
519  T q = C / A;
520  T D = p * p - q;
521 
522  if (!approxEquals(D, (T)0))
523  {
524  if (D < (T)0)
525  return 0;
526 
527  T sqrtD = sqrt(D);
528  roots[0] = sqrtD - p;
529  roots[1] = -sqrtD - p;
530 
531  return 2;
532  }
533  else
534  {
535  roots[0] = -p;
536  roots[1] = -p;
537 
538  return 1;
539  }
540  }
541  else
542  {
543  return solveLinear(B, C, roots);
544  }
545  }

559  template <typename T>
560  static UINT32 solveCubic(T A, T B, T C, T D, T* roots)
561  {
562  static const T THIRD = (1 / (T)3);
563 
564  T invA = 1 / A;
565  A = B * invA;
566  B = C * invA;
567  C = D * invA;
568 
569  T sqA = A * A;
570  T p = THIRD * (-THIRD * sqA + B);
571  T q = ((T)0.5) * ((2 / (T)27) * A * sqA - THIRD * A * B + C);
572 
573  T cbp = p * p * p;
574  D = q * q + cbp;
575 
576  UINT32 numRoots = 0;
577  if (!approxEquals(D, (T)0))
578  {
579  if (D < 0.0)
580  {
581  T phi = THIRD * ::acos(-q / sqrt(-cbp));
582  T t = 2 * sqrt(-p);
583 
584  roots[0] = t * cos(phi);
585  roots[1] = -t * cos(phi + PI * THIRD);
586  roots[2] = -t * cos(phi - PI * THIRD);
587 
588  numRoots = 3;
589  }
590  else
591  {
592  T sqrtD = sqrt(D);
593  T u = cbrt(sqrtD + fabs(q));
594 
595  if (q > (T)0)
596  roots[0] = -u + p / u;
597  else
598  roots[0] = u - p / u;
599 
600  numRoots = 1;
601  }
602  }
603  else
604  {
605  if (!approxEquals(q, (T)0))
606  {
607  T u = cbrt(-q);
608  roots[0] = 2 * u;
609  roots[1] = -u;
610 
611  numRoots = 2;
612  }
613  else
614  {
615  roots[0] = 0.0f;
616  numRoots = 1;
617  }
618  }
619 
620  T sub = THIRD * A;
621  for (UINT32 i = 0; i < numRoots; i++)
622  roots[i] -= sub;
623 
624  return numRoots;
625  }

640  template <typename T>
641  static UINT32 solveQuartic(T A, T B, T C, T D, T E, T* roots)
642  {
643  T invA = 1 / A;
644  A = B * invA;
645  B = C * invA;
646  C = D * invA;
647  D = E * invA;
648 
649  T sqA = A*A;
650  T p = -(3 / (T)8) * sqA + B;
651  T q = (1 / (T)8) * sqA * A - (T)0.5 * A * B + C;
652  T r = -(3 / (T)256) * sqA * sqA + (1 / (T)16) * sqA * B - (1 / (T)4) * A * C + D;
653 
654  UINT32 numRoots = 0;
655  if (!approxEquals(r, (T)0))
656  {
657  T cubicA = 1;
658  T cubicB = -(T)0.5 * p ;
659  T cubicC = -r;
660  T cubicD = (T)0.5 * r * p - (1 / (T)8) * q * q;
661 
662  solveCubic(cubicA, cubicB, cubicC, cubicD, roots);
663  T z = roots[0];
664 
665  T u = z * z - r;
666  T v = 2 * z - p;
667 
668  if (approxEquals(u, T(0)))
669  u = 0;
670  else if (u > 0)
671  u = sqrt(u);
672  else
673  return 0;
674 
675  if (approxEquals(v, T(0)))
676  v = 0;
677  else if (v > 0)
678  v = sqrt(v);
679  else
680  return 0;
681 
682  T quadraticA = 1;
683  T quadraticB = q < 0 ? -v : v;
684  T quadraticC = z - u;
685 
686  numRoots = solveQuadratic(quadraticA, quadraticB, quadraticC, roots);
687 
688  quadraticA = 1;
689  quadraticB = q < 0 ? v : -v;
690  quadraticC = z + u;
691 
692  numRoots += solveQuadratic(quadraticA, quadraticB, quadraticC, roots + numRoots);
693  }
694  else
695  {
696  numRoots = solveCubic(q, p, (T)0, (T)1, roots);
697  roots[numRoots++] = 0;
698  }
699 
700  T sub = (1/(T)4) * A;
701  for (UINT32 i = 0; i < numRoots; i++)
702  roots[i] -= sub;
703 
704  return numRoots;
705  }

717  template<class T>
718  static T cubicHermite(float t, const T& pointA, const T& pointB, const T& tangentA, const T& tangentB)
719  {
720  float t2 = t * t;
721  float t3 = t2 * t;
722 
723  float a = 2 * t3 - 3 * t2 + 1;
724  float b = t3 - 2 * t2 + t;
725  float c = -2 * t3 + 3 * t2;
726  float d = t3 - t2;
727 
728  return a * pointA + b * tangentA + c * pointB + d * tangentB;
729  }

788  template<class T>
789  static void cubicHermiteCoefficients(const T& pointA, const T& pointB, const T& tangentA, const T& tangentB,
790  float length, T (&coefficients)[4])
791  {
792  float length2 = length * length;
793  float invLength2 = 1.0f / length2;
794  float invLength3 = 1.0f / (length2 * length);
795 
796  T scaledTangentA = tangentA * length;
797  T scaledTangentB = tangentB * length;
798 
799  T diff = pointA - pointB;
800 
801  coefficients[0] = (2 * diff + scaledTangentA + scaledTangentB) * invLength3;
802  coefficients[1] = (-3 * diff - 2 * scaledTangentA - scaledTangentB) * invLength2;
803  coefficients[2] = tangentA;
804  coefficients[3] = pointA;
805  }
