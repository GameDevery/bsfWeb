454  template <typename T>
455  static UINT32 solveQuadratic(T A, T B, T C, T* roots)
456  {
457  if (!approxEquals(A, (T)0))
458  {
459  T p = B / (2 * A);
460  T q = C / A;
461  T D = p * p - q;
462 
463  if (!approxEquals(D, (T)0))
464  {
465  if (D < (T)0)
466  return 0;
467 
468  T sqrtD = sqrt(D);
469  roots[0] = sqrtD - p;
470  roots[1] = -sqrtD - p;
471 
472  return 2;
473  }
474  else
475  {
476  roots[0] = -p;
477  roots[1] = -p;
478 
479  return 1;
480  }
481  }
482  else
483  {
484  return solveLinear(B, C, roots);
485  }
486  }

500  template <typename T>
501  static UINT32 solveCubic(T A, T B, T C, T D, T* roots)
502  {
503  static const T THIRD = (1 / (T)3);
504 
505  T invA = 1 / A;
506  A = B * invA;
507  B = C * invA;
508  C = D * invA;
509 
510  T sqA = A * A;
511  T p = THIRD * (-THIRD * sqA + B);
512  T q = ((T)0.5) * ((2 / (T)27) * A * sqA - THIRD * A * B + C);
513 
514  T cbp = p * p * p;
515  D = q * q + cbp;
516 
517  UINT32 numRoots = 0;
518  if (!approxEquals(D, (T)0))
519  {
520  if (D < 0.0)
521  {
522  T phi = THIRD * ::acos(-q / sqrt(-cbp));
523  T t = 2 * sqrt(-p);
524 
525  roots[0] = t * cos(phi);
526  roots[1] = -t * cos(phi + PI * THIRD);
527  roots[2] = -t * cos(phi - PI * THIRD);
528 
529  numRoots = 3;
530  }
531  else
532  {
533  T sqrtD = sqrt(D);
534  T u = cbrt(sqrtD + fabs(q));
535 
536  if (q > (T)0)
537  roots[0] = -u + p / u;
538  else
539  roots[0] = u - p / u;
540 
541  numRoots = 1;
542  }
543  }
544  else
545  {
546  if (!approxEquals(q, (T)0))
547  {
548  T u = cbrt(-q);
549  roots[0] = 2 * u;
550  roots[1] = -u;
551 
552  numRoots = 2;
553  }
554  else
555  {
556  roots[0] = 0.0f;
557  numRoots = 1;
558  }
559  }
560 
561  T sub = THIRD * A;
562  for (UINT32 i = 0; i < numRoots; i++)
563  roots[i] -= sub;
564 
565  return numRoots;
566  }

581  template <typename T>
582  static UINT32 solveQuartic(T A, T B, T C, T D, T E, T* roots)
583  {
584  T invA = 1 / A;
585  A = B * invA;
586  B = C * invA;
587  C = D * invA;
588  D = E * invA;
589 
590  T sqA = A*A;
591  T p = -(3 / (T)8) * sqA + B;
592  T q = (1 / (T)8) * sqA * A - (T)0.5 * A * B + C;
593  T r = -(3 / (T)256) * sqA * sqA + (1 / (T)16) * sqA * B - (1 / (T)4) * A * C + D;
594 
595  UINT32 numRoots = 0;
596  if (!approxEquals(r, (T)0))
597  {
598  T cubicA = 1;
599  T cubicB = -(T)0.5 * p ;
600  T cubicC = -r;
601  T cubicD = (T)0.5 * r * p - (1 / (T)8) * q * q;
602 
603  solveCubic(cubicA, cubicB, cubicC, cubicD, roots);
604  T z = roots[0];
605 
606  T u = z * z - r;
607  T v = 2 * z - p;
608 
609  if (approxEquals(u, T(0)))
610  u = 0;
611  else if (u > 0)
612  u = sqrt(u);
613  else
614  return 0;
615 
616  if (approxEquals(v, T(0)))
617  v = 0;
618  else if (v > 0)
619  v = sqrt(v);
620  else
621  return 0;
622 
623  T quadraticA = 1;
624  T quadraticB = q < 0 ? -v : v;
625  T quadraticC = z - u;
626 
627  numRoots = solveQuadratic(quadraticA, quadraticB, quadraticC, roots);
628 
629  quadraticA = 1;
630  quadraticB = q < 0 ? v : -v;
631  quadraticC = z + u;
632 
633  numRoots += solveQuadratic(quadraticA, quadraticB, quadraticC, roots + numRoots);
634  }
635  else
636  {
637  numRoots = solveCubic(q, p, (T)0, (T)1, roots);
638  roots[numRoots++] = 0;
639  }
640 
641  T sub = (1/(T)4) * A;
642  for (UINT32 i = 0; i < numRoots; i++)
643  roots[i] -= sub;
644 
645  return numRoots;
646  }

658  template<class T>
659  static T cubicHermite(float t, const T& pointA, const T& pointB, const T& tangentA, const T& tangentB)
660  {
661  float t2 = t * t;
662  float t3 = t2 * t;
663 
664  float a = 2 * t3 - 3 * t2 + 1;
665  float b = t3 - 2 * t2 + t;
666  float c = -2 * t3 + 3 * t2;
667  float d = t3 - t2;
668 
669  return a * pointA + b * tangentA + c * pointB + d * tangentB;
670  }

729  template<class T>
730  static void cubicHermiteCoefficients(const T& pointA, const T& pointB, const T& tangentA, const T& tangentB,
731  float length, T (&coefficients)[4])
732  {
733  float length2 = length * length;
734  float invLength2 = 1.0f / length2;
735  float invLength3 = 1.0f / (length2 * length);
736 
737  T scaledTangentA = tangentA * length;
738  T scaledTangentB = tangentB * length;
739 
740  T diff = pointA - pointB;
741 
742  coefficients[0] = (2 * diff + scaledTangentA + scaledTangentB) * invLength3;
743  coefficients[1] = (-3 * diff - 2 * scaledTangentA - scaledTangentB) * invLength2;
744  coefficients[2] = tangentA;
745  coefficients[3] = pointA;
746  }
