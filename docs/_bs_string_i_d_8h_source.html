138  static void toMemory(const StringID& data, char* memory)
139  {
140  UINT32 size = getDynamicSize(data);
141 
142  UINT32 curSize = sizeof(UINT32);
143  memcpy(memory, &size, curSize);
144  memory += curSize;
145 
146  bool isEmpty = data.empty();
147  memory = rttiWriteElem(isEmpty, memory);
148 
149  if (!isEmpty)
150  {
151  UINT32 length = (UINT32)strlen(data.cstr());
152  memcpy(memory, data.cstr(), length * sizeof(char));
153  }
154  }

156  static UINT32 fromMemory(StringID& data, char* memory)
157  {
158  UINT32 size;
159  memcpy(&size, memory, sizeof(UINT32));
160  memory += sizeof(UINT32);
161 
162  bool empty = false;
163  memory = rttiReadElem(empty, memory);
164 
165  if (!empty)
166  {
167  UINT32 length = (size - sizeof(UINT32) - sizeof(bool)) / sizeof(char);
168 
169  auto name = (char*)bs_stack_alloc(length + 1);
170  memcpy(name, memory, length);
171  name[length] = '\0';
172 
173  data = StringID(name);
174  bs_stack_free(name);
175  }
176 
177  return size;
178  }
179 
180  static UINT32 getDynamicSize(const StringID& data)
181  {
182  UINT32 dataSize = sizeof(bool) + sizeof(UINT32);
183 
184  bool isEmpty = data.empty();
185  if (!isEmpty)
186  {
187  UINT32 length = (UINT32)strlen(data.cstr());
188  dataSize += length * sizeof(char);
189  }
190 
191  return (UINT32)dataSize;
192  }
