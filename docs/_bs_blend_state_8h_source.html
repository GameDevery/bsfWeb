266  {
267  UINT32 size;
268  memcpy(&size, memory, sizeof(UINT32));
269  memory += sizeof(UINT32);
270 
271  UINT32 dataSize = size - sizeof(UINT32);
272  memcpy((void*)&data, memory, dataSize);
273 
274  return size;
275  }

278  {
