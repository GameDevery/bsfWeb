31 inline void get_cpuid(unsigned level, unsigned subleaf, unsigned* eax, unsigned* ebx,
32  unsigned* ecx, unsigned* edx)
33 {
34 #if __clang__ || (__INTEL_COMPILER && !_MSC_VER)
35  // Older versions of clang don't support subleafs, which leads to inability
36  // to detect AVX2 for example. On ICC there's no proper cpuid intrinsic.
37 #if SIMDPP_32_BITS
38  __asm("cpuid" : "=a"(*eax), "=b" (*ebx), "=c"(*ecx), "=d"(*edx) \
39  : "0"(level), "2"(subleaf));
40 #else
41  // x86-64 uses %rbx as the base register, so preserve it.
42  __asm("xchgq %%rbx,%q1\n" \
43  "cpuid\n" \
44  "xchgq %%rbx,%q1" \
45  : "=a"(*eax), "=b" (*ebx), "=c"(*ecx), "=d"(*edx) \
46  : "0"(level), "2"(subleaf));
47 #endif
48 #elif __GNUC__
49  __cpuid_count(level, subleaf, *eax, *ebx, *ecx, *edx);
50 #elif _MSC_VER
51  uint32_t regs[4];
52  __cpuidex((int*) regs, level, subleaf);
53  *eax = regs[0];
54  *ebx = regs[1];
55  *ecx = regs[2];
56  *edx = regs[3];
57 #else
58  #error "unsupported compiler"
59 #endif
60 }

62 inline uint64_t get_xcr(unsigned level)
63 {
64 #if (defined (_MSC_FULL_VER) && _MSC_FULL_VER >= 160040000) || \
65  (defined (__INTEL_COMPILER) && __INTEL_COMPILER >= 1200)
66  return _xgetbv(level);
67 #elif defined(__GNUC__) || defined(__clang__)
68  uint32_t eax, edx;
69  __asm("xgetbv" : "=a"(eax),"=d"(edx) : "c"(level) : );
70  return eax | (uint64_t(edx) << 32);
71 #else
72  return 0;
73 #endif
74 }

82 static inline cpu_manufacturer
83  get_cpu_manufacturer(uint32_t ebx, uint32_t ecx, uint32_t edx)
84 {
85  if (ebx == 0x756E6547 && ecx == 0x6C65746E && edx == 0x49656E69)
86  return CPU_INTEL; // "GenuineIntel"
87  if (ebx == 0x68747541 && ecx == 0x444D4163 && edx == 0x69746E65)
88  return CPU_AMD; // "AuthenticAMD"
89  return CPU_UNKNOWN;
90 }

97 inline Arch get_arch_raw_cpuid()
98 {
99  Arch arch_info = Arch::NONE_NULL;
100 
101  uint32_t eax, ebx, ecx, edx;
102  bool xsave_xrstore_avail = false;
103 
104  simdpp::detail::get_cpuid(0, 0, &eax, &ebx, &ecx, &edx);
105  unsigned max_cpuid_level = eax;
106  simdpp::detail::cpu_manufacturer mfg =
107  simdpp::detail::get_cpu_manufacturer(ebx, ecx, edx);
108 
109  simdpp::detail::get_cpuid(0x80000000, 0, &eax, &ebx, &ecx, &edx);
110  unsigned max_ex_cpuid_level = eax;
111 
112  if (max_cpuid_level >= 0x00000001) {
113  simdpp::detail::get_cpuid(0x00000001, 0, &eax, &ebx, &ecx, &edx);
114 
115  if (edx & (1u << 26))
116  arch_info |= Arch::X86_SSE2;
117  if (ecx & (1u << 0))
118  arch_info |= Arch::X86_SSE3;
119  if (ecx & (1u << 9))
120  arch_info |= Arch::X86_SSSE3;
121  if (ecx & (1u << 19))
122  arch_info |= Arch::X86_SSE4_1;
123  if (ecx & (1u << 20) && mfg == simdpp::detail::CPU_INTEL)
124  arch_info |= Arch::X86_POPCNT_INSN; // popcnt is included in SSE4.2 on Intel
125  if (ecx & (1u << 23))
126  arch_info |= Arch::X86_POPCNT_INSN;
127  if (ecx & (1u << 12))
128  arch_info |= Arch::X86_FMA3;
129  if (ecx & (1u << 26)) {
130  // XSAVE/XRSTORE available on hardware, now check OS support
131  uint64_t xcr = simdpp::detail::get_xcr(0);
132  if ((xcr & 6) == 6)
133  xsave_xrstore_avail = true;
134  }
135 
136  if (ecx & (1u << 28) && xsave_xrstore_avail)
137  arch_info |= Arch::X86_AVX;
138  }
139  if (max_ex_cpuid_level >= 0x80000001) {
140  simdpp::detail::get_cpuid(0x80000001, 0, &eax, &ebx, &ecx, &edx);
141  if (ecx & (1u << 16))
142  arch_info |= Arch::X86_FMA4;
143  if (ecx & (1u << 11))
144  arch_info |= Arch::X86_XOP;
145  }
146 
147  if (max_cpuid_level >= 0x00000007) {
148  simdpp::detail::get_cpuid(0x00000007, 0, &eax, &ebx, &ecx, &edx);
149  if (ebx & (1u << 5) && xsave_xrstore_avail)
150  arch_info |= Arch::X86_AVX2;
151  if (ebx & (1u << 16) && xsave_xrstore_avail)
152  arch_info |= Arch::X86_AVX512F;
153  if (ebx & (1u << 30) && xsave_xrstore_avail)
154  arch_info |= Arch::X86_AVX512BW;
155  if (ebx & (1u << 17) && xsave_xrstore_avail)
156  arch_info |= Arch::X86_AVX512DQ;
157  if (ebx & (1u << 31) && xsave_xrstore_avail)
158  arch_info |= Arch::X86_AVX512VL;
159  }
160 
161  return arch_info;
162 }
