40  template<class T>
41  struct RTTIPlainType
42  {
43  static_assert(std::is_pod<T>::value,
44  "Provided type isn't plain-old-data. You need to specialize RTTIPlainType template in order to serialize this type. "\
45  " (Or call BS_ALLOW_MEMCPY_SERIALIZATION(type) macro if you are sure the type can be properly serialized using just memcpy.)");

165 #define BS_ALLOW_MEMCPY_SERIALIZATION(type) \
166  static_assert (std::is_trivially_copyable<type>()==true, \
167  #type " is not trivially copyable"); \
168  template<> struct RTTIPlainType<type> \
169  { enum { id=0 }; enum { hasDynamicSize = 0 }; \
170  static void toMemory(const type& data, char* memory) \
171  { memcpy(memory, &data, sizeof(type)); } \
172  static UINT32 fromMemory(type& data, char* memory) \
173  { memcpy(&data, memory, sizeof(type)); return sizeof(type); } \
174  static UINT32 getDynamicSize(const type& data) \
175  { return sizeof(type); } \
176  };

185  template<class T> struct RTTIPlainType<std::vector<T, StdAlloc<T>>>
186  {
187  enum { id = TID_Vector }; enum { hasDynamicSize = 1 };
188 
190  static void toMemory(const std::vector<T, StdAlloc<T>>& data, char* memory)
191  {
192  UINT32 size = sizeof(UINT32);
193  char* memoryStart = memory;
194  memory += sizeof(UINT32);
195 
196  UINT32 numElements = (UINT32)data.size();
197  memcpy(memory, &numElements, sizeof(UINT32));
198  memory += sizeof(UINT32);
199  size += sizeof(UINT32);
200 
201  for(const auto& item : data)
202  {
203  UINT32 elementSize = rttiGetElemSize(item);
204  RTTIPlainType<T>::toMemory(item, memory);
205 
206  memory += elementSize;
207  size += elementSize;
208  }
209 
210  memcpy(memoryStart, &size, sizeof(UINT32));
211  }
212 
214  static UINT32 fromMemory(std::vector<T, StdAlloc<T>>& data, char* memory)
215  {
216  UINT32 size = 0;
217  memcpy(&size, memory, sizeof(UINT32));
218  memory += sizeof(UINT32);
219 
220  UINT32 numElements;
221  memcpy(&numElements, memory, sizeof(UINT32));
222  memory += sizeof(UINT32);
223 
224  for(UINT32 i = 0; i < numElements; i++)
225  {
226  T element;
227  UINT32 elementSize = RTTIPlainType<T>::fromMemory(element, memory);
228  data.push_back(element);
229 
230  memory += elementSize;
231  }
232 
233  return size;
234  }
235 
237  static UINT32 getDynamicSize(const std::vector<T, StdAlloc<T>>& data)
238  {
239  UINT64 dataSize = sizeof(UINT32) * 2;
240 
241  for(const auto& item : data)
242  dataSize += rttiGetElemSize(item);
243 
244  assert(dataSize <= std::numeric_limits<UINT32>::max());
245 
246  return (UINT32)dataSize;
247  }
248  };
