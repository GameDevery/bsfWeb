195  T* allocate(const size_t num) const
196  {
197  if (num == 0)
198  return nullptr;
199 
200  if (num > static_cast<size_t>(-1) / sizeof(T))
201  return nullptr; // Error
202 
203  void* const pv = mFrameAlloc->alloc((UINT32)(num * sizeof(T)));
204  if (!pv)
205  return nullptr; // Error
206 
207  return static_cast<T*>(pv);
208  }
