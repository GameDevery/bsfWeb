281  static constexpr RGBA lerp(UINT8 t, RGBA from, RGBA to)
282  {
283  constexpr UINT32 RB_MASK = 0x00FF00FF;
284  constexpr UINT32 GA_MASK = 0xFF00FF00;
285 
286  // Lerp two channels at a time (this leaves 8 extra bits for each channel for results)
288  const UINT32 rbFrom = from & RB_MASK;
289  const UINT32 rbTo = to & RB_MASK;
290  const UINT32 rb = (rbFrom + (((rbTo - rbFrom) * t) >> 8)) & RB_MASK;
291 
293  const UINT32 gaFrom = from & GA_MASK;
294  const UINT32 gaTo = to & GA_MASK;
295  const UINT32 ga = (gaFrom + ((((gaTo >> 8) - (gaFrom >> 8)) * t) >> 8)) & GA_MASK;
296 
297  return rb | ga;
298  }
299
