112  {
113  if (mask == 0)
114  return 0;

125  template<typename SrcT, typename DestT>
126  static DestT convertBitPattern(SrcT srcValue, SrcT srcBitMask, DestT destBitMask)
127  {
128  // Mask off irrelevant source value bits (if any)
129  srcValue = srcValue & srcBitMask;
130 
131  // Shift source down to bottom of DWORD
132  const unsigned int srcBitShift = getBitShift(srcBitMask);
133  srcValue >>= srcBitShift;
134 
135  // Get max value possible in source from srcMask
136  const SrcT srcMax = srcBitMask >> srcBitShift;
137 
138  // Get max available in dest
139  const unsigned int destBitShift = getBitShift(destBitMask);
140  const DestT destMax = destBitMask >> destBitShift;
141 
142  // Scale source value into destination, and shift back
143  DestT destValue = (srcValue * destMax) / srcMax;
144  return (destValue << destBitShift);
145  }

151  static unsigned int fixedToFixed(UINT32 value, unsigned int n, unsigned int p)
152  {
153  if (n > p)
154  {
155  // Less bits required than available; this is easy
156  value >>= n - p;
157  }
158  else if (n < p)
159  {
160  // More bits required than are there, do the fill
161  // Use old fashioned division, probably better than a loop
162  if (value == 0)
163  value = 0;
164  else if (value == (static_cast<unsigned int>(1) << n) - 1)
165  value = (1 << p) - 1;
166  else value = value*(1 << p) / ((1 << n) - 1);
167  }
168  return value;
169  }

205  {
206  switch(n) {

216  ((UINT8*)dest)[1] = (UINT8)((value >> 8) & 0xFF);
217  ((UINT8*)dest)[2] = (UINT8)(value & 0xFF);
218 #else
219  ((UINT8*)dest)[2] = (UINT8)((value >> 16) & 0xFF);
220  ((UINT8*)dest)[1] = (UINT8)((value >> 8) & 0xFF);
221  ((UINT8*)dest)[0] = (UINT8)(value & 0xFF);
222 #endif
223  break;
224  case 4:
225  ((UINT32*)dest)[0] = (UINT32)value;
226  break;
227  }

262  static UINT16 floatToHalfI(UINT32 i)
263  {
264  int s = (i >> 16) & 0x00008000;
265  int e = ((i >> 23) & 0x000000ff) - (127 - 15);
266  int m = i & 0x007fffff;
267 
268  if (e <= 0)
269  {
270  if (e < -10)
271  {
272  return 0;
273  }
274  m = (m | 0x00800000) >> (1 - e);
275 
276  return static_cast<UINT16>(s | (m >> 13));
277  }
278  else if (e == 0xff - (127 - 15))
279  {
280  if (m == 0) // Inf
281  {
282  return static_cast<UINT16>(s | 0x7c00);
283  }
284  else // NAN
285  {
286  m >>= 13;
287  return static_cast<UINT16>(s | 0x7c00 | m | (m == 0));
288  }
289  }
290  else
291  {
292  if (e > 30) // Overflow
293  {
294  return static_cast<UINT16>(s | 0x7c00);
295  }
296 
297  return static_cast<UINT16>(s | (e << 10) | (m >> 13));
298  }
299  }

310  static UINT32 halfToFloatI(UINT16 y)
311  {
312  int s = (y >> 15) & 0x00000001;
313  int e = (y >> 10) & 0x0000001f;
314  int m = y & 0x000003ff;
315 
316  if (e == 0)
317  {
318  if (m == 0) // Plus or minus zero
319  {
320  return s << 31;
321  }
322  else // Denormalized number -- renormalize it
323  {
324  while (!(m & 0x00000400))
325  {
326  m <<= 1;
327  e -= 1;
328  }
329 
330  e += 1;
331  m &= ~0x00000400;
332  }
333  }
334  else if (e == 31)
335  {
336  if (m == 0) // Inf
337  {
338  return (s << 31) | 0x7f800000;
339  }
340  else // NaN
341  {
342  return (s << 31) | 0x7f800000 | (m << 13);
343  }
344  }
345 
346  e = e + (127 - 15);
347  m = m << 13;
348 
349  return (s << 31) | (e << 23) | m;
350  }

353  static UINT32 floatToFloat10(float v)
354  {
355  Float754 f;
356  f.value = v;
357 
358  if (f.field.exponent == 0xFF)
359  {
360  // NAN or INF
361  if (f.field.mantissa > 0)
362  return 0x3E0 | (((f.raw >> 18) | (f.raw >> 13) | (f.raw >> 3) | f.raw) & 0x1F);
363  else if (f.field.negative)
364  return 0; // Negative infinity clamped to 0
365  else
366  return 0x3E0; // Positive infinity
367 
368  }
369  else if (f.field.negative)
370  return 0; // Negative clamped to 0, no negatives allowed
371  else if (f.raw > 0x477C0000)
372  return 0x3DF; // Too large, clamp to max value
373  else
374  {
375  UINT32 val;
376  if (f.raw < 0x38800000U)
377  {
378  // Too small to be represented as a normalized float, convert to denormalized value
379  UINT32 shift = 113 - f.field.exponent;
380  val = (0x800000U | f.field.mantissa) >> shift;
381  }
382  else
383  {
384  // Rebias exponent
385  val = f.raw + 0xC8000000;
386  }
387 
388  return ((val + 0x1FFFFU + ((val >> 18) & 1)) >> 18) & 0x3FF;
389  }
390  }

433  static float float10ToFloat(UINT32 v)
434  {
435  Float10 f;
436  f.raw = v;
437 
438  UINT32 output;
439  if (f.field.exponent == 0x1F) // INF or NAN
440  {
441  output = 0x7f800000 | (f.field.mantissa << 17);
442  }
443  else
444  {
445  UINT32 exponent;
446  UINT32 mantissa = f.field.mantissa;
447 
448  if (f.field.exponent != 0) // The value is normalized
449  exponent = f.field.exponent;
450  else if (mantissa != 0) // The value is denormalized
451  {
452  // Normalize the value in the resulting float
453  exponent = 1;
454 
455  do
456  {
457  exponent--;
458  mantissa <<= 1;
459  } while ((mantissa & 0x20) == 0);
460 
461  mantissa &= 0x1F;
462  }
463  else // The value is zero
464  exponent = (UINT32)-112;
465 
466  output = ((exponent + 112) << 23) | (mantissa << 18);
467  }
468 
469  return *(float*)&output;
470  }
