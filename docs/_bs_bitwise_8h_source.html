126  template<typename SrcT, typename DestT>
127  static DestT convertBitPattern(SrcT srcValue, SrcT srcBitMask, DestT destBitMask)
128  {
129  // Mask off irrelevant source value bits (if any)
130  srcValue = srcValue & srcBitMask;
131 
132  // Shift source down to bottom of DWORD
133  const uint32_t srcBitShift = getBitShift(srcBitMask);
134  srcValue >>= srcBitShift;
135 
136  // Get max value possible in source from srcMask
137  const SrcT srcMax = srcBitMask >> srcBitShift;
138 
139  // Get max available in dest
140  const uint32_t destBitShift = getBitShift(destBitMask);
141  const DestT destMax = destBitMask >> destBitShift;
142 
143  // Scale source value into destination, and shift back
144  DestT destValue = (srcValue * destMax) / srcMax;
145  return (destValue << destBitShift);
146  }

152  static uint32_t fixedToFixed(UINT32 value, uint32_t n, uint32_t p)
153  {
154  if (n > p)
155  {
156  // Less bits required than available; this is easy
157  value >>= n - p;
158  }
159  else if (n < p)
160  {
161  // More bits required than are there, do the fill
162  // Use old fashioned division, probably better than a loop
163  if (value == 0)
164  value = 0;
165  else if (value == (static_cast<uint32_t>(1) << n) - 1)
166  value = (1 << p) - 1;
167  else value = value*(1 << p) / ((1 << n) - 1);
168  }
169  return value;
170  }

359  static UINT16 floatToHalfI(UINT32 i)
360  {
361  int32_t s = (i >> 16) & 0x00008000;
362  int32_t e = ((i >> 23) & 0x000000ff) - (127 - 15);
363  int32_t m = i & 0x007fffff;
364 
365  if (e <= 0)
366  {
367  if (e < -10)
368  {
369  return 0;
370  }
371  m = (m | 0x00800000) >> (1 - e);
372 
373  return static_cast<UINT16>(s | (m >> 13));
374  }
375  else if (e == 0xff - (127 - 15))
376  {
377  if (m == 0) // Inf
378  {
379  return static_cast<UINT16>(s | 0x7c00);
380  }
381  else // NAN
382  {
383  m >>= 13;
384  return static_cast<UINT16>(s | 0x7c00 | m | (m == 0));
385  }
386  }
387  else
388  {
389  if (e > 30) // Overflow
390  {
391  return static_cast<UINT16>(s | 0x7c00);
392  }
393 
394  return static_cast<UINT16>(s | (e << 10) | (m >> 13));
395  }
396  }

407  static UINT32 halfToFloatI(UINT16 y)
408  {
409  int32_t s = (y >> 15) & 0x00000001;
410  int32_t e = (y >> 10) & 0x0000001f;
411  int32_t m = y & 0x000003ff;
412 
413  if (e == 0)
414  {
415  if (m == 0) // Plus or minus zero
416  {
417  return s << 31;
418  }
419  else // Denormalized number -- renormalize it
420  {
421  while (!(m & 0x00000400))
422  {
423  m <<= 1;
424  e -= 1;
425  }
426 
427  e += 1;
428  m &= ~0x00000400;
429  }
430  }
431  else if (e == 31)
432  {
433  if (m == 0) // Inf
434  {
435  return (s << 31) | 0x7f800000;
436  }
437  else // NaN
438  {
439  return (s << 31) | 0x7f800000 | (m << 13);
440  }
441  }
442 
443  e = e + (127 - 15);
444  m = m << 13;
445 
446  return (s << 31) | (e << 23) | m;
447  }

450  static UINT32 floatToFloat10(float v)
451  {
452  Float754 f;
453  f.value = v;
454 
455  if (f.field.exponent == 0xFF)
456  {
457  // NAN or INF
458  if (f.field.mantissa > 0)
459  return 0x3E0 | (((f.raw >> 18) | (f.raw >> 13) | (f.raw >> 3) | f.raw) & 0x1F);
460  else if (f.field.negative)
461  return 0; // Negative infinity clamped to 0
462  else
463  return 0x3E0; // Positive infinity
464 
465  }
466  else if (f.field.negative)
467  return 0; // Negative clamped to 0, no negatives allowed
468  else if (f.raw > 0x477C0000)
469  return 0x3DF; // Too large, clamp to max value
470  else
471  {
472  UINT32 val;
473  if (f.raw < 0x38800000U)
474  {
475  // Too small to be represented as a normalized float, convert to denormalized value
476  UINT32 shift = 113 - f.field.exponent;
477  val = (0x800000U | f.field.mantissa) >> shift;
478  }
479  else
480  {
481  // Rebias exponent
482  val = f.raw + 0xC8000000;
483  }
484 
485  return ((val + 0x1FFFFU + ((val >> 18) & 1)) >> 18) & 0x3FF;
486  }
487  }

530  static float float10ToFloat(UINT32 v)
531  {
532  Float10 f;
533  f.raw = v;
534 
535  UINT32 output;
536  if (f.field.exponent == 0x1F) // INF or NAN
537  {
538  output = 0x7f800000 | (f.field.mantissa << 17);
539  }
540  else
541  {
542  UINT32 exponent;
543  UINT32 mantissa = f.field.mantissa;
544 
545  if (f.field.exponent != 0) // The value is normalized
546  exponent = f.field.exponent;
547  else if (mantissa != 0) // The value is denormalized
548  {
549  // Normalize the value in the resulting float
550  exponent = 1;
551 
552  do
553  {
554  exponent--;
555  mantissa <<= 1;
556  } while ((mantissa & 0x20) == 0);
557 
558  mantissa &= 0x1F;
559  }
560  else // The value is zero
561  exponent = (UINT32)-112;
562 
563  output = ((exponent + 112) << 23) | (mantissa << 18);
564  }
565 
566  return *(float*)&output;
567  }
