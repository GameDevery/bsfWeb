29  template <int ElemSize, int ElemsPerBlock = 512, int Alignment = 4>
30  class PoolAlloc
31  {
32  private:
34  class MemBlock
35  {
36  public:
37  MemBlock(UINT8* blockData)
38  :blockData(blockData), freePtr(0), freeElems(ElemsPerBlock), nextBlock(nullptr)
39  {
40  UINT32 offset = 0;
41  for(UINT32 i = 0; i < ElemsPerBlock; i++)
42  {
43  UINT32* entryPtr = (UINT32*)&blockData[offset];
44 
45  offset += ActualElemSize;
46  *entryPtr = offset;
47  }
48  }
49 
50  ~MemBlock()
51  {
52  assert(freeElems == ElemsPerBlock && "Not all elements were deallocated from a block.");
53  }
54 
59  UINT8* alloc()
60  {
61  UINT8* freeEntry = &blockData[freePtr];
62  freePtr = *(UINT32*)freeEntry;
63  --freeElems;
64 
65  return freeEntry;
66  }
67 
69  void dealloc(void* data)
70  {
71  UINT32* entryPtr = (UINT32*)data;
72  *entryPtr = freePtr;
73  ++freeElems;
74 
75  freePtr = (UINT32)(((UINT8*)data) - blockData);
76  }
77 
78  UINT8* blockData;
79  UINT32 freePtr;
80  UINT32 freeElems;
81  MemBlock* nextBlock;
82  };

84  public:
85  PoolAlloc()
86  {
87  static_assert(ElemSize >= 4, "Pool allocator minimum allowed element size is 4 bytes.");
88  static_assert(ElemsPerBlock > 0, "Number of elements per block must be at least 1.");
89  static_assert(ElemsPerBlock * ActualElemSize <= UINT_MAX, "Pool allocator block size too large.");
90  }

92  ~PoolAlloc()
93  {
94  MemBlock* curBlock = mFreeBlock;
95  while (curBlock != nullptr)
96  {
97  MemBlock* nextBlock = curBlock->nextBlock;
98  deallocBlock(curBlock);
99 
100  curBlock = nextBlock;
101  }
102  }

105  UINT8* alloc()
106  {
107  if(mFreeBlock == nullptr || mFreeBlock->freeElems == 0)
108  allocBlock();
109 
110  mTotalNumElems++;
111  return mFreeBlock->alloc();
112  }
113 
115  void free(void* data)
116  {
117  MemBlock* curBlock = mFreeBlock;
118  while(curBlock)
119  {
120  constexpr UINT32 blockDataSize = ActualElemSize * ElemsPerBlock;
121  if(data >= curBlock->blockData && data < (curBlock->blockData + blockDataSize))
122  {
123  curBlock->dealloc(data);
124  mTotalNumElems--;
125 
126  if(curBlock->freeElems == 0 && curBlock->nextBlock)
127  {
128  // Free the block, but only if there is some extra free space in other blocks
129  const UINT32 totalSpace = (mNumBlocks - 1) * ElemsPerBlock;
130  const UINT32 freeSpace = totalSpace - mTotalNumElems;
131 
132  if(freeSpace > ElemsPerBlock / 2)
133  {
134  mFreeBlock = curBlock->nextBlock;
135  deallocBlock(curBlock);
136  }
137  }
138 
139  return;
140  }
141 
142  curBlock = curBlock->nextBlock;
143  }
144 
145  assert(false);
146  }

166  private:
168  MemBlock* allocBlock()
169  {
170  MemBlock* newBlock = nullptr;
171  MemBlock* curBlock = mFreeBlock;
172 
173  while (curBlock != nullptr)
174  {
175  MemBlock* nextBlock = curBlock->nextBlock;
176  if (nextBlock != nullptr && nextBlock->freeElems > 0)
177  {
178  // Found an existing block with free space
179  newBlock = nextBlock;
180 
181  curBlock->nextBlock = newBlock->nextBlock;
182  newBlock->nextBlock = mFreeBlock;
183 
184  break;
185  }
186 
187  curBlock = nextBlock;
188  }
189 
190  if (newBlock == nullptr)
191  {
192  constexpr UINT32 blockDataSize = ActualElemSize * ElemsPerBlock;
193  size_t paddedBlockDataSize = blockDataSize + (Alignment - 1); // Padding for potential alignment correction
194 
195  UINT8* data = (UINT8*)bs_alloc(sizeof(MemBlock) + (UINT32)paddedBlockDataSize);
196 
197  void* blockData = data + sizeof(MemBlock);
198  blockData = std::align(Alignment, blockDataSize, blockData, paddedBlockDataSize);
199 
200  newBlock = new (data) MemBlock((UINT8*)blockData);
201  mNumBlocks++;
202 
203  newBlock->nextBlock = mFreeBlock;
204  }
205 
206  mFreeBlock = newBlock;
207  return newBlock;
208  }
209 
211  void deallocBlock(MemBlock* block)
212  {
213  block->~MemBlock();
214  bs_free(block);
215 
216  mNumBlocks--;
217  }
218 
219  static constexpr int ActualElemSize = ((ElemSize + Alignment - 1) / Alignment) * Alignment;
220 
221  MemBlock* mFreeBlock = nullptr;
222  UINT32 mTotalNumElems = 0;
223  UINT32 mNumBlocks = 0;
224  };
