15 /* The values of the tags are such that it is easy to retrieve the result type
16  of an operation that involves two different types. That is, if the
17  arguments have types with type tags A and B then the type tag of the result
18  type will be max(A, B). Similarly, if the arguments have types with size
19  tags P and Q then the size tag of the result type will be max(P, Q).
20 
21  The only exception is operation on masks with masks with different size
22  tags. In that case the type tag of the result type is always TAG_UINT.
23 
24  The following list illustrates the conversion rules. The first two items
25  are the argument types and the third item is the result type.
26 
27  [ Same element sizes ]
28 
29  intXX | intXX -> intXX
30  intXX | uintXX -> uintXX
31  uintXX | uintXX -> uintXX
32  floatXX | intXX -> floatXX
33  floatXX | uintXX -> floatXX
34  floatXX | floatXX -> floatXX
35  mask_intXX | intXX -> uintXX
36  mask_intXX | uintXX -> uintXX
37  mask_intXX | mask_intXX -> mask_intXX
38  mask_intXX | floatXX -> floatXX
39  mask_intXX | mask_floatXX -> mask_floatXX
40  mask_floatXX | floatXX -> floatXX
41  mask_floatXX | mask_floatXX -> mask_floatXX
42 
43  [ Different element sizes ]
44 
45  intXX | intYY -> int{ max(XX, YY) }
46  intXX | uintYY -> uint{ max(XX, YY) }
47  uintXX | uintYY -> uint{ max(XX, YY) }
48  floatXX | intYY -> float{ max(XX, YY) }
49  floatXX | uintYY -> float{ max(XX, YY) }
50  floatXX | floatYY -> float{ max(XX, YY) }
51  mask_intXX | intYY -> uint{ max(XX, YY) }
52  mask_intXX | uintYY -> uint{ max(XX, YY) }
53  mask_intXX | mask_intYY -> uint{ max(XX, YY) }
54  mask_intXX | floatYY -> float{ max(XX, YY) }
55  mask_intXX | mask_floatYY -> uint{ max(XX, YY) }
56  mask_floatXX | floatYY -> float{ max(XX, YY) }
57  mask_floatXX | mask_floatYY -> uint{ max(XX, YY) }
58 
59  See also simdpp/detail/get_expr.h
60 
61  If particular operation does not support masks, the masks are expanded
62  internally. If both arguments are masks, then the result type is the
63  underlying type of the mask type which would be retrieved using the usual
64  rules (see get_expr2_nomask).
65 
66  Several examples:
67 
68  mask_int32 + mask_int32 -> uint32
69  int32 + mask_int32 -> int32
70  uint32 + mask_int32 -> uint32
71 */
72 #define SIMDPP_TAG_MASK_INT 1
73 #define SIMDPP_TAG_MASK_FLOAT 2
74 #define SIMDPP_TAG_INT 3
75 #define SIMDPP_TAG_UINT 4
76 #define SIMDPP_TAG_FLOAT 5
77 #define SIMDPP_TAG_SIZE8 10
78 #define SIMDPP_TAG_SIZE16 20
79 #define SIMDPP_TAG_SIZE32 30
80 #define SIMDPP_TAG_SIZE64 40
81 
82 #endif
