26  void setSeed(uint32_t seed)
27  {
28  mSeed[0] = seed;
29  mSeed[1] = seed * 434521248 + 1; // Arbitrary random numbers
30  mSeed[2] = seed * 854624531 + 1;
31  mSeed[3] = seed * 694295795 + 1;
32  }

35  uint32_t get() const
36  {
37  // Using xorshift128 algorithm
38  uint32_t t = mSeed[3];
39  t ^= t << 11;
40  t ^= t >> 8;
41 
42  mSeed[3] = mSeed[2];
43  mSeed[2] = mSeed[1];
44  mSeed[1] = mSeed[0];
45 
46  const uint32_t s = mSeed[0];
47  t ^= s;
48  t ^= s >> 19;
49 
50  mSeed[0] = t;
51  return t;
52  }

55  int32_t getRange(int32_t min, int32_t max) const
56  {
57  assert(max > min);
58 
59  // Note: Not using modulo for performance
60  const int32_t range = max - min + 1;
61 
62  constexpr static float DELTA = 0e-5f;
63  return min + (int32_t)(getUNorm() * ((float)range - DELTA));
64  }

80  Vector3 getUnitVector() const
81  {
82  // Pick a random number on a unit cube and use the result only if squared size less than 1. This is faster
83  // than most other methods, and generally not many iterations are required to get the required vector.
84 
85  Vector3 output;
86  float sqrdSize;
87 
88  do
89  {
90  output.x = getSNorm();
91  output.y = getSNorm();
92  output.z = getSNorm();
93  sqrdSize = output.squaredLength();
94 
95  } while (sqrdSize > 1.0f || sqrdSize < 0.001f);
96 
97  return Vector3::normalize(output);
98  }

186  Vector3 getBarycentric() const
187  {
188  float u = getUNorm();
189  float v = getUNorm();
190 
191  if((u + v) > 1.0f)
192  {
193  u = 1.0f - u;
194  v = 1.0f - v;
195  }
196 
197  const float w = 1.0f - u - v;
198  return Vector3(u, v, w);
