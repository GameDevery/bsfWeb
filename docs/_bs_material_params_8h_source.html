73  template <typename T>
74  void getDataParam(const String& name, UINT32 arrayIdx, T& output) const
75  {
76  GpuParamDataType dataType = TGpuDataParamInfo<T>::TypeId;
77 
78  const ParamData* param = nullptr;
79  auto result = getParamData(name, ParamType::Data, dataType, arrayIdx, &param);
80  if (result != GetParamResult::Success)
81  return;
82 
83  const GpuParamDataTypeInfo& typeInfo = GpuParams::PARAM_SIZES.lookup[dataType];
84  UINT32 paramTypeSize = typeInfo.numColumns * typeInfo.numRows * typeInfo.baseTypeSize;
85  output = *(T*)(mDataParamsBuffer[param->index + arrayIdx * paramTypeSize]);
86 
87  memcpy(output, &mDataParamsBuffer[param->index + arrayIdx * paramTypeSize], sizeof(paramTypeSize));
88  }

183  {
184  GpuParamDataType dataType = (GpuParamDataType)TGpuDataParamInfo<T>::TypeId;
185 
186  const GpuParamDataTypeInfo& typeInfo = GpuParams::PARAM_SIZES.lookup[dataType];

198  template <typename T>
199  void setDataParam(const ParamData& param, UINT32 arrayIdx, const T& input) const
200  {
201  GpuParamDataType dataType = (GpuParamDataType)TGpuDataParamInfo<T>::TypeId;
202 
203  const GpuParamDataTypeInfo& typeInfo = GpuParams::PARAM_SIZES.lookup[dataType];
204  UINT32 paramTypeSize = typeInfo.numColumns * typeInfo.numRows * typeInfo.baseTypeSize;
205 
206  assert(sizeof(input) == paramTypeSize);
207  memcpy(&mDataParamsBuffer[param.index + arrayIdx * paramTypeSize], &input, paramTypeSize);
