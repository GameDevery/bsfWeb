23 namespace simdpp {
24 
29 inline Arch get_arch_linux_cpuinfo()
30 {
31  Arch res = Arch::NONE_NULL;
32 
33  std::map<std::string, Arch> features;
34  std::string ident;
35 
36 #if __arm__
37  ident = "Features\t";
38  features["neon"] = Arch::ARM_NEON | Arch::ARM_NEON_FLT_SP;
39 
40 #elif __aarch64__
41  ident = "Features\t";
42  features["asimd"] = Arch::ARM_NEON | Arch::ARM_NEON_FLT_SP;
43 
44 #elif __i386__ || __amd64__
45  Arch a_sse2 = Arch::X86_SSE2;
46  Arch a_sse3 = a_sse2 | Arch::X86_SSE3;
47  Arch a_ssse3 = a_sse3 | Arch::X86_SSSE3;
48  Arch a_sse4_1 = a_ssse3 | Arch::X86_SSE4_1;
49  Arch a_popcnt = Arch::X86_POPCNT_INSN;
50  Arch a_avx = a_sse4_1 | Arch::X86_AVX;
51  Arch a_avx2 = a_avx | Arch::X86_AVX2;
52  Arch a_fma3 = a_sse3 | Arch::X86_FMA3;
53  Arch a_fma4 = a_sse3 | Arch::X86_FMA4;
54  Arch a_xop = a_sse3 | Arch::X86_XOP;
55  Arch a_avx512f = a_avx2 | Arch::X86_AVX512F;
56  Arch a_avx512bw = a_avx512f | Arch::X86_AVX512BW;
57  Arch a_avx512dq = a_avx512f | Arch::X86_AVX512DQ;
58  Arch a_avx512vl = a_avx512f | Arch::X86_AVX512VL;
59 
60  ident = "flags\t";
61  features["sse2"] = a_sse2;
62  features["pni"] = a_sse3;
63  features["ssse3"] = a_ssse3;
64  features["sse4_1"] = a_sse4_1;
65  features["avx"] = a_avx;
66  features["avx2"] = a_avx2;
67  features["popcnt"] = a_popcnt;
68  features["fma"] = a_fma3;
69  features["fma4"] = a_fma4;
70  features["xop"] = a_xop;
71  features["avx512f"] = a_avx512f;
72  features["avx512bw"] = a_avx512bw;
73  features["avx512dq"] = a_avx512dq;
74  features["avx512vl"] = a_avx512vl;
75 #else
76  return res;
77 #endif
78 
79  std::ifstream in("/proc/cpuinfo");
80 
81  if (!in) {
82  return res;
83  }
84 
85  std::string line;
86  while (std::getline(in, line)) {
87  // Check whether identification string matches
88  if (line.length() < ident.length()) {
89  continue;
90  }
91  auto r = std::mismatch(ident.begin(), ident.end(), line.begin());
92  if (r.first != ident.end()) {
93  continue;
94  }
95 
96  // Get the items
97  std::istringstream sin(std::string(r.second, line.end()));
98  std::vector<std::string> items;
99 
100  std::copy(std::istream_iterator<std::string>(sin),
101  std::istream_iterator<std::string>(),
102  std::back_inserter(items));
103 
104  // Match items to known features
105  for (auto& item : items) {
106  auto it = features.find(item);
107  if (it == features.end()) {
108  continue;
109  }
110  res |= it->second;
111  }
112  }
113 
114  return res;
115 }
116 
117 } // namespace simdpp
118 
119 #endif
120 #endif
