204  template<class T>
205  Vector<GameObjectHandle<T>> SceneManager::findComponents(bool activeOnly)
206  {
207  UINT32 rttiId = T::getRTTIStatic()->getRTTIId();
208 
209  Vector<GameObjectHandle<T>> output;
210  for(auto& entry : mActiveComponents)
211  {
212  if (isComponentOfType(entry, rttiId))
213  output.push_back(static_object_cast<T>(entry));
214  }
215 
216  if(!activeOnly)
217  {
218  for(auto& entry : mInactiveComponents)
219  {
220  if (isComponentOfType(entry, rttiId))
221  output.push_back(static_object_cast<T>(entry));
222  }
223 
224  for(auto& entry : mUninitializedComponents)
225  {
226  if (isComponentOfType(entry, rttiId))
227  output.push_back(static_object_cast<T>(entry));
228  }
229  }
230 
231  return output;
232  }
