57  template <class T>
58  SpriteMaterial* registerMaterial()
59  {
60  SpriteMaterial* newMaterial = bs_new<T>();
61 
62  UINT32 id = newMaterial->getId();
63  auto iterFind = mMaterials.find(id);
64  if (iterFind != mMaterials.end())
65  {
66  // Already exists
67  LOGWRN("Attempting to register a sprite material that already exists, ignoring request.");
68  bs_delete(newMaterial);
69  return iterFind->second;
70  }
71 
72  mMaterials[id] = newMaterial;
73  return newMaterial;
74  }
75  private:
76  UnorderedMap<UINT32, SpriteMaterial*> mMaterials;
