145  class ParticleSet : public INonCopyable
146  {
148  static constexpr float CAPACITY_SCALE = 1.2f; // 20%
149 
150  public:
155  ParticleSet(UINT32 capacity)
156  :mParticles(capacity)
157  { }
158 
164  UINT32 allocParticles(UINT32 count)
165  {
166  const UINT32 particleIdx = mCount;
167  mCount += count;
168 
169  if(mCount > mParticles.capacity)
170  {
171  const auto newCapacity = (UINT32)(mCount * CAPACITY_SCALE);
172  ParticleSetData newData(newCapacity, mParticles);
173  mParticles = std::move(newData);
174  }
175 
176  const UINT32 particleEnd = particleIdx + count;
177  if(particleEnd > mMaxIndex)
178  {
179  for (; mMaxIndex < particleEnd; mMaxIndex++)
180  mParticles.indices[mMaxIndex] = mMaxIndex;
181  }
182 
183  return particleIdx;
184  }
185 
187  void freeParticle(UINT32 idx)
188  {
189  // Note: We always keep the active particles sequential. This makes it faster to iterate over all particles, but
190  // increases the cost when removing particles. Considering iteration should happen many times per-particle,
191  // while removal will happen only once, this should be the more performant approach, but will likely be worth
192  // profiling in the future. An alternative approach is to flag dead particles without moving them.
193 
194  assert(idx < mCount);
195 
196  const UINT32 lastIdx = mCount - 1;
197  if(idx != lastIdx)
198  {
199  std::swap(mParticles.position[idx], mParticles.position[lastIdx]);
200  std::swap(mParticles.velocity[idx], mParticles.velocity[lastIdx]);
201  std::swap(mParticles.size[idx], mParticles.size[lastIdx]);
202  std::swap(mParticles.rotation[idx], mParticles.rotation[lastIdx]);
203  std::swap(mParticles.lifetime[idx], mParticles.lifetime[lastIdx]);
204  std::swap(mParticles.color[idx], mParticles.color[lastIdx]);
205  std::swap(mParticles.indices[idx], mParticles.indices[lastIdx]);
206  }
207 
208  mCount--;
209  }
210 
212  void clear(UINT32 numPartices = 0)
213  {
214  if(mCount > numPartices)
215  mCount = numPartices;
216  }

231  UINT32 determineTextureSize() const

235  UINT32 width = Bitwise::nextPow2(count);

240  width /= 2;
241  height *= 2;
242  }
243 
244  // Make it square
