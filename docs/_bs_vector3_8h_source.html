353  static void orthonormalize(Vector3& vec0, Vector3& vec1, Vector3& vec2)
354  {
355  vec0.normalize();
356 
357  float dot0 = vec0.dot(vec1);
358  vec1 -= dot0*vec0;
359  vec1.normalize();
360
