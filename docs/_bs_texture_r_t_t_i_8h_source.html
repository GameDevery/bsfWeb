36  INT32& getUsage(Texture* obj) { return obj->mProperties.mDesc.usage; }
37  void setUsage(Texture* obj, INT32& val)
38  {
39  // Render target and depth stencil texture formats are for in-memory use only
40  // and don't make sense when serialized
41  if ((val & (TU_DEPTHSTENCIL | TU_RENDERTARGET)) != 0)
42  {
43  obj->mProperties.mDesc.usage &= ~(TU_DEPTHSTENCIL | TU_RENDERTARGET);
44  obj->mProperties.mDesc.usage |= TU_STATIC;
45  }
46  else
47  obj->mProperties.mDesc.usage = val;
48  }

53  SPtr<PixelData> getPixelData(Texture* obj, UINT32 idx)
54  {
55  UINT32 face = (size_t)Math::floor(idx / (float)(obj->mProperties.getNumMipmaps() + 1));
56  UINT32 mipmap = idx % (obj->mProperties.getNumMipmaps() + 1);
57 
58  SPtr<PixelData> pixelData = obj->mProperties.allocBuffer(face, mipmap);
59 
60  obj->readData(pixelData, face, mipmap);
61  gCoreThread().submitAll(true);
62 
63  return pixelData;
64  }

94  void onDeserializationStarted(IReflectable* obj, const UnorderedMap<String, UINT64>& params) override
95  {
96  Texture* texture = static_cast<Texture*>(obj);
97 
98  texture->mRTTIData = bs_new<Vector<SPtr<PixelData>>>();
99  }
100 
101  void onDeserializationEnded(IReflectable* obj, const UnorderedMap<String, UINT64>& params) override
102  {
103  Texture* texture = static_cast<Texture*>(obj);
104 
105  if(texture->mRTTIData.empty())
106  return;
107 
108  TextureProperties& texProps = texture->mProperties;
109 
110  // Update pixel format if needed as it's possible the original texture was saved using some other render API
111  // that has an unsupported format.
112  PixelFormat originalFormat = texProps.getFormat();
113  PixelFormat validFormat = TextureManager::instance().getNativeFormat(
114  texProps.getTextureType(), texProps.getFormat(), texProps.getUsage(), texProps.isHardwareGammaEnabled());
115 
116  Vector<SPtr<PixelData>>* pixelData = any_cast<Vector<SPtr<PixelData>>*>(texture->mRTTIData);
117  if (originalFormat != validFormat)
118  {
119  texProps.mDesc.format = validFormat;
120 
121  for (size_t i = 0; i < pixelData->size(); i++)
122  {
123  SPtr<PixelData> origData = pixelData->at(i);
124  SPtr<PixelData> newData = PixelData::create(origData->getWidth(), origData->getHeight(), origData->getDepth(), validFormat);
125 
126  PixelUtil::bulkPixelConversion(*origData, *newData);
127  (*pixelData)[i] = newData;
128  }
129  }
130 
131  // A bit clumsy initializing with already set values, but I feel its better than complicating things and storing the values
132  // in mRTTIData.
133  texture->initialize();
134 
135  for(size_t i = 0; i < pixelData->size(); i++)
136  {
137  UINT32 face = (size_t)Math::floor(i / (float)(texProps.getNumMipmaps() + 1));
138  UINT32 mipmap = i % (texProps.getNumMipmaps() + 1);
139 
140  texture->writeData(pixelData->at(i), face, mipmap, false);
141  }
142 
143  bs_delete(pixelData);
144  texture->mRTTIData = nullptr;
145  }
