154  SIMDPP_INL float64<2> unmask() const
155  {

168 #endif

170  SIMDPP_INL const mask_float64<2>& vec(unsigned) const { return *this; }
171  SIMDPP_INL mask_float64<2>& vec(unsigned) { return *this; }
172 
173  SIMDPP_INL mask_float64<2> eval() const { return *this; }
174 
175 private:
176  native_type d_;
177 };
