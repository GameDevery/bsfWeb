57  void fromString(const String& versionString)
58  {
59  Vector<bs::String> tokens = StringUtil::split(versionString, ".");
60  if(!tokens.empty())
61  {
62  major = parseINT32(tokens[0]);
63  if (tokens.size() > 1)
64  minor = parseINT32(tokens[1]);
65  if (tokens.size() > 2)
66  release = parseINT32(tokens[2]);
67  if (tokens.size() > 3)
68  build = parseINT32(tokens[3]);
69  }
70 
71  }
72 
73  INT32 major = 0;
74  INT32 minor = 0;
75  INT32 release = 0;
76  INT32 build = 0;
77  };

238  void setCapability(const Capabilities c)
239  {
240  UINT64 index = (CAPS_CATEGORY_MASK & c) >> BS_CAPS_BITSHIFT;
241  mCapabilities[index] |= (c & ~CAPS_CATEGORY_MASK);
242  }
243 
245  void unsetCapability(const Capabilities c)
246  {
247  UINT64 index = (CAPS_CATEGORY_MASK & c) >> BS_CAPS_BITSHIFT;
248  mCapabilities[index] &= (~c | CAPS_CATEGORY_MASK);
249  }
250 
252  bool hasCapability(const Capabilities c) const
253  {
254  UINT64 index = (CAPS_CATEGORY_MASK & c) >> BS_CAPS_BITSHIFT;
255 
256  return (mCapabilities[index] & (c & ~CAPS_CATEGORY_MASK)) != 0;
257  }
