147  template<class T>
148  static void registerNodeType()
149  {
150  auto findIter = mNodeTypes.find(T::getNodeId());
151  if (findIter != mNodeTypes.end())
152  LOGERR("Found two render compositor nodes with the same name \"" + String(T::getNodeId().cstr()) + "\".");
153 
154  mNodeTypes[T::getNodeId()] = bs_new<TNodeType<T>>();
155  }
156 
158  static void cleanUp()
159  {
160  for (auto& entry : mNodeTypes)
161  bs_delete(entry.second);
162 
163  mNodeTypes.clear();
164  }
165 
166  private:
167  static UnorderedMap<StringID, NodeType*> mNodeTypes;
168  };
