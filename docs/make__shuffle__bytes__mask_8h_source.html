117 template<int s0, int s1, unsigned N> SIMDPP_INL
118 uint8<N> make_shuffle_bytes16_mask(uint8<N> &mask)
119 {
120  detail::assert_selector_range<s0,s1,2>();
121  const unsigned b0 = detail::get_shuffle_bytex1_16<s0,2>::r0;
122  const unsigned b1 = detail::get_shuffle_bytex1_16<s1,2>::r0;
123  mask = make_uint(b0, b1, b0+2, b1+2,
124  b0+4, b1+4, b0+6, b1+6,
125  b0+8, b1+8, b0+10,b1+10,
126  b0+12,b1+12,b0+14,b1+14);
127  return mask;
128 }
